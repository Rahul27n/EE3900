\section{QUESTION: GATE EC 2005 Q.85}
A non-zero sequence $x(n)$ is given by:
\begin{figure}[!ht]
    \centering
    \includegraphics[width=0.5\columnwidth] {Gate_Assignment_3_Fig_1.png}
    \caption{$x(n)$}
    \label{x(n)}
\end{figure}
\\The sequence
\begin{align}
y(n)=  
\begin{cases}
x\Bigg(\dfrac{n}{2}-1\Bigg), & \text{for } n \text{ even}\\
0, & \text{for } n \text{ odd}\nonumber
\end{cases}
\end{align}
is given by:
\begin{figure}[!ht]
    \centering
    \includegraphics[width=\columnwidth] {Gate_Assignment_3_Fig_2.png}
    \caption{Options}
    \label{Options}
\end{figure}
\section{SOLUTION}
We can write $x(n)$ as follows:
\begin{align}
x(n) &= \frac{1}{2}\delta(n+2) + \delta(n+1) + 2\delta(n) + \delta(n-1) + \frac{1}{2}\delta(n-2) \nonumber
\end{align}
We clearly have:
\begin{align}
y(-2) &= x(-2) = \frac{1}{2}\\
y(0) &= x(-1) = 1\\
y(2) &= x(0) = 2\\
y(4) &= x(1) = 1\\
y(6) &= x(2) = \frac{1}{2}
\end{align}
We also have :
\begin{align}
y(-1) &= y(1) = y(3) = y(5) = 0
\end{align}
Therefore we can write $y(n)$ as follows:
\begin{align}
y(n) &= \frac{1}{2}\delta(n+2) + \delta(n) + 2\delta(n-2) + \delta(n-4) + \frac{1}{2}\delta(n-6) \nonumber
\end{align}
Hence the correct answer is option (a).
